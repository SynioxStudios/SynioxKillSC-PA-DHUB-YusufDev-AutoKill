-- ULTRA HIZLI GLOBAL KILL SİSTEMİ
task.spawn(function()
    while _G.KillActive do
        pcall(function()
            local char = LocalPlayer.Character
            local punch = LocalPlayer.Backpack:FindFirstChild("Punch") or (char and char:FindFirstChild("Punch"))
            
            if punch then
                if punch.Parent ~= char then punch.Parent = char end
                
                -- BURST ATTACK: Her döngüde 10 kat vuruş sinyali gönderir
                for i = 1, 10 do
                    LocalPlayer.muscleEvent:FireServer("punch", "rightHand")
                    LocalPlayer.muscleEvent:FireServer("punch", "leftHand")
                end
                
                local rHand = char:FindFirstChild("RightHand")
                if rHand then
                    for _, target in ipairs(Players:GetPlayers()) do
                        if target ~= LocalPlayer and target.Character then
                            local tRoot = target.Character:FindFirstChild("HumanoidRootPart")
                            local tHum = target.Character:FindFirstChild("Humanoid")
                            
                            -- Mesafe sınırı olmadan dokunma sinyali (Saniyede onlarca kez)
                            if tRoot and tHum and tHum.Health > 0 then
                                firetouchinterest(rHand, tRoot, 1)
                                firetouchinterest(rHand, tRoot, 0)
                            end
                        end
                    end
                end
            end
        end)
        -- task.wait() kaldırıldı veya minimuma çekildi (En yüksek hız)
        RS.Heartbeat:Wait() 
    end
end)

-- VERİ GÜNCELLEME
local frames = 0
local lastUpdate = tick()
RS.RenderStepped:Connect(function()
    frames = frames + 1
    if tick() - lastUpdate >= 1 then
        FPSLabel.Text = "FPS: " .. frames
        frames = 0
        lastUpdate = tick()
        local stats = LocalPlayer:FindFirstChild("leaderstats") and LocalPlayer.leaderstats:FindFirstChild("Kills")
        KillLabel.Text = "Kills: " .. (stats and tostring(stats.Value) or "0")
    end
end)

-- SERVER HOPPER
task.spawn(function()
    local t = _G.HopInterval
    while t > 0 do
        TimerLabel.Text = "Hop: " .. t .. "s"
        task.wait(1)
        t = t - 1
    end
    
    local function findServer()
        local success, res = pcall(function()
            return HttpService:JSONDecode(game:HttpGet("https://games.roblox.com/v1/games/"..game.PlaceId.."/servers/Public?sortOrder=Desc&limit=50"))
        end)
        if success and res.data then
            for _, s in pairs(res.data) do
                if s.playing < (s.maxPlayers - 2) and s.id ~= game.JobId then
                    TeleportService:TeleportToPlaceInstance(game.PlaceId, s.id, LocalPlayer)
                    return
                end
            end
        end
        task.wait(2)
        findServer()
    end
    findServer()
end)

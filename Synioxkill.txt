-- KILL & DATA DÖNGÜSÜ
task.spawn(function()
    while true do
        if _G.KillActive then
            pcall(function()
                local char = LocalPlayer.Character
                local punch = LocalPlayer.Backpack:FindFirstChild("Punch") or (char and char:FindFirstChild("Punch"))
                if punch and punch.Parent ~= char then punch.Parent = char end
                
                LocalPlayer.muscleEvent:FireServer("punch", "rightHand")
                LocalPlayer.muscleEvent:FireServer("punch", "leftHand")

                for _, v in ipairs(Players:GetPlayers()) do
                    if v ~= LocalPlayer and v.Character and v.Character:FindFirstChild("HumanoidRootPart") then
                        local root = v.Character.HumanoidRootPart
                        local hum = v.Character:FindFirstChild("Humanoid")
                        if hum and hum.Health > 0 and char:FindFirstChild("RightHand") then
                            firetouchinterest(char.RightHand, root, 1)
                            firetouchinterest(char.RightHand, root, 0)
                        end
                    end
                end
            end)
        end
        task.wait(0.01)
    end
end)

-- SAYAÇLAR
local frameCount = 0
local lastUpdate = tick()
RS.RenderStepped:Connect(function()
    frameCount = frameCount + 1
    if tick() - lastUpdate >= 1 then
        FPSLabel.Text = "FPS: " .. frameCount
        frameCount = 0
        lastUpdate = tick()
        local kills = LocalPlayer:FindFirstChild("leaderstats") and LocalPlayer.leaderstats:FindFirstChild("Kills")
        KillLabel.Text = "Kills: " .. (kills and tostring(kills.Value) or "0")
    end
end)

-- SERVER HOP
task.spawn(function()
    local t = _G.HopInterval
    while t > 0 do
        TimerLabel.Text = "Hop: " .. t .. "s"
        task.wait(1)
        t = t - 1
    end
    local res = HttpService:JSONDecode(game:HttpGet("https://games.roblox.com/v1/games/"..game.PlaceId.."/servers/Public?sortOrder=Desc&limit=50"))
    for _, s in pairs(res.data) do
        if s.playing < (s.maxPlayers - 2) and s.id ~= game.JobId then
            TeleportService:TeleportToPlaceInstance(game.PlaceId, s.id, LocalPlayer)
        end
    end
end)

-- DURMADAN VURMA VE HEDEFLEME SİSTEMİ
RunService.Stepped:Connect(function()
    if _G.KillActive then
        pcall(function()
            for _, v in pairs(game.Players:GetPlayers()) do
                if v ~= lp and v.Character and v.Character:FindFirstChild("Humanoid") and v.Character.Humanoid.Health > 0 then
                    -- Durmadan vuruş sinyali (Burst miktarını 15'e çıkardım)
                    for i = 1, 15 do
                        rs.rEvents.punchEvent:FireServer("punchClick")
                    end
                end
            end
        end)
    end
end)

-- FPS VE KILL GÜNCELLEME
local frameCount = 0
local lastUpdate = tick()
RunService.RenderStepped:Connect(function()
    frameCount = frameCount + 1
    if tick() - lastUpdate >= 1 then
        FPSLabel.Text = "FPS: " .. frameCount
        frameCount = 0
        lastUpdate = tick()
        
        -- Kill sayısını her saniye kontrol et
        local kills = lp:FindFirstChild("leaderstats") and lp.leaderstats:FindFirstChild("Kills")
        if kills then KillLabel.Text = "Kills: " .. kills.Value end
    end
end)

-- SERVER HOP (45 Saniye)
spawn(function()
    local timeLeft = _G.HopInterval
    while timeLeft > 0 do
        TimerLabel.Text = "Hop: " .. timeLeft .. "s"
        task.wait(1)
        timeLeft = timeLeft - 1
    end
    
    -- Sunucu Değiştirme İşlemi
    local api = "https://games.roblox.com/v1/games/" .. game.PlaceId .. "/servers/Public?sortOrder=Desc&limit=100"
    local s, res = pcall(function() return HttpService:JSONDecode(game:HttpGet(api)) end)
    if s then
        for _, server in pairs(res.data) do
            if server.playing < server.maxPlayers and server.id ~= game.JobId then
                TeleportService:TeleportToPlaceInstance(game.PlaceId, server.id, lp)
            end
        end
    end
end)

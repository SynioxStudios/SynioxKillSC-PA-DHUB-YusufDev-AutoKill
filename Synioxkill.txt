-- KATLİAM DÖNGÜSÜ
task.spawn(function()
    while true do
        if _G.KillActive then
            pcall(function()
                local char = LocalPlayer.Character
                local punch = LocalPlayer.Backpack:FindFirstChild("Punch") or (char and char:FindFirstChild("Punch"))
                
                if punch and punch.Parent ~= char then punch.Parent = char end

                -- Hızlı Remote Vuruş
                LocalPlayer.muscleEvent:FireServer("punch", "rightHand")
                LocalPlayer.muscleEvent:FireServer("punch", "leftHand")

                -- Alan Hasarı
                local rHand = char:FindFirstChild("RightHand")
                local lHand = char:FindFirstChild("LeftHand")

                if rHand and lHand then
                    for _, target in ipairs(Players:GetPlayers()) do
                        if target ~= LocalPlayer and not playerWhitelist[target.Name] then
                            local tChar = target.Character
                            local tRoot = tChar and tChar:FindFirstChild("HumanoidRootPart")
                            local tHum = tChar and tChar:FindFirstChild("Humanoid")

                            if tRoot and tHum and tHum.Health > 0 then
                                firetouchinterest(rHand, tRoot, 1)
                                firetouchinterest(rHand, tRoot, 0)
                                firetouchinterest(lHand, tRoot, 1)
                                firetouchinterest(lHand, tRoot, 0)
                            end
                        end
                    end
                end
            end)
        end
        task.wait(0.01)
    end
end)

-- FPS & KILL UPDATE
local frameCount = 0
local lastUpdate = tick()
RS.RenderStepped:Connect(function()
    frameCount = frameCount + 1
    if tick() - lastUpdate >= 1 then
        FPSLabel.Text = "FPS: " .. frameCount
        frameCount = 0
        lastUpdate = tick()
        local kills = LocalPlayer:FindFirstChild("leaderstats") and LocalPlayer.leaderstats:FindFirstChild("Kills")
        KillLabel.Text = "Kills: " .. (kills and tostring(kills.Value) or "0")
    end
end)

-- SERVER HOPPER (Boş Server Garantili)
local function serverHop()
    local api = "https://games.roblox.com/v1/games/" .. game.PlaceId .. "/servers/Public?sortOrder=Desc&limit=100"
    local s, res = pcall(function() return HttpService:JSONDecode(game:HttpGet(api)) end)
    if s and res.data then
        for _, server in pairs(res.data) do
            if server.playing < (server.maxPlayers - 2) and server.id ~= game.JobId then
                TeleportService:TeleportToPlaceInstance(game.PlaceId, server.id, LocalPlayer)
                return
            end
        end
    end
    task.wait(2)
    serverHop()
end

-- 45 SN SAYACI
task.spawn(function()
    local timeLeft = _G.HopInterval
    while timeLeft > 0 do
        TimerLabel.Text = "Hop: " .. timeLeft .. "s"
        task.wait(1)
        timeLeft = timeLeft - 1
    end
    serverHop()
end)
